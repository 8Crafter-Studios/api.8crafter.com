Interface RegExpAccessibleCommandType

The type of command data with an accessType of regexp.

    accessType: "regexp";

        event: ChatSendBeforeEvent,
    ) => CommandResponse;
    categories?: [commandCategory?, ...commandCategory[]];
    command_version?: string;

    flagsDocs?: string;
    formatting_code?: string;
    functional?: boolean;
    hidden?: boolean;
    requiredTags?: string[];
    syntax?: string;
    type: "built-in" | "custom" | "unknown";

        | "WorldEdit"
        | "owner"
        | "headAdmin"
        | "admin"
